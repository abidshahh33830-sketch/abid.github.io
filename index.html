let tasks = JSON.parse(localStorage.getItem("tasks")) || []; // Page load par tasks show karo window.onload = function () { showTasks(); }; function addTask() { let taskInput = document.getElementById("taskInput"); let taskValue = taskInput.value; if (taskValue === "") { alert("Please enter a task!"); return; } tasks.push({ text: taskValue, completed: false }); localStorage.setItem("tasks", JSON.stringify(tasks)); taskInput.value = ""; showTasks(); } function showTasks() { let taskList = document.getElementById("taskList"); taskList.innerHTML = ""; tasks.forEach((task, index) => { let li = document.createElement("li"); li.innerHTML = ` ${task.text} `; taskList.appendChild(li); }); } function deleteTask(index) { tasks.splice(index, 1); localStorage.setItem("tasks", JSON.stringify(tasks)); showTasks(); } function toggleComplete(index) { tasks[index].completed = !tasks[index].completed; localStorage.setItem("tasks", JSON.stringify(tasks)); showTasks(); }
